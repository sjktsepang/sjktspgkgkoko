// ── LocalStorage helpers ───────────────────────────────────────────────────
function loadRecords(){
  try{
    const raw=localStorage.getItem(LS_KEY);
    return raw?JSON.parse(raw):[];
  }catch(e){return[]}
}
function saveRecords(recs){
  try{localStorage.setItem(LS_KEY,JSON.stringify(recs))}catch(e){}
}

// ── Helpers ────────────────────────────────────────────────────────────────
function padZ(n){return String(n).padStart(2,"0")}
function fmtTime(d){return`${padZ(d.getHours())}:${padZ(d.getMinutes())}:${padZ(d.getSeconds())}`}
function fmtTimeSh(d){return`${padZ(d.getHours())}:${padZ(d.getMinutes())}`}
function fmtDate(d){
  const days=["Ahad","Isnin","Selasa","Rabu","Khamis","Jumaat","Sabtu"];
  const months=["Jan","Feb","Mac","Apr","Mei","Jun","Jul","Ogo","Sep","Okt","Nov","Dis"];
  return`${days[d.getDay()]}, ${d.getDate()} ${months[d.getMonth()]} ${d.getFullYear()}`;
}
function isoDate(d){return`${d.getFullYear()}-${padZ(d.getMonth()+1)}-${padZ(d.getDate())}`}

function useRealTime(){
  const[now,setNow]=useState(new Date());
  useEffect(()=>{const t=setInterval(()=>setNow(new Date()),1000);return()=>clearInterval(t)},[]);
  return now;
}
